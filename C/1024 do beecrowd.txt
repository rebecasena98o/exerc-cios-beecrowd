#include <stdio.h>
#include <string.h>
#include <ctype.h>

int main() {
    int N;
    char texto[1001];

    scanf("%d\n", &N);

    while (N--) {
        fgets(texto, 1001, stdin);
        texto[strcspn(texto, "\n")] = '\0';

        int len = strlen(texto);

        // 1ª passada: desloca letras
        for (int i = 0; i < len; i++) {
            if (isalpha(texto[i]))
                texto[i] += 3;
        }

        // 2ª passada: inverter
        for (int i = 0; i < len / 2; i++) {
            char tmp = texto[i];
            texto[i] = texto[len - 1 - i];
            texto[len - 1 - i] = tmp;
        }

        // 3ª passada: -1 na metade final
        for (int i = len / 2; i < len; i++) {
            texto[i] -= 1;
        }

        printf("%s\n", texto);
    }

    return 0;
}
